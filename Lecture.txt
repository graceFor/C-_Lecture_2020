// assign_fun.c
#include <stdio.h>

void assign(int *x, int y)
{
    *x = y;
}
int main()
{
    int a = 0, b = 999;
    assign(&a, b);
    printf("%d\n", a);
    return 0;
}

//pswap.c
#include <stdio.h>

void pswap(int *a, int *b)
{
    int temp;
    temp = *a;
    *a = *b;
    *b = temp;
}
int main()
{
    int one, two;
    printf("one과 two의 값을 입력하시오: ");
    scanf("%d %d", &one, &two);
    pswap(&one, &two);
    printf("%d %d", one, two);
}

//포인터를 배열처럼 사용하기
#include <stdio.h>

int main()
{
    char str[6] = "hello";
    char *p;

    p = str;

    printf("%s\n", str);        // hello
    printf("%s\n", p);          // hello
    printf("%c\n", str[1]);     // e
    printf("%c\n", p[1]);       // e
    printf("%c\n", *(str + 1)); // e
    printf("%c\n", *(p + 1));   // e
}

//포인터를 배열처럼 사용하기
#include <stdio.h>

int main()
{
    char name[20] = "Jessie Norman";

    first = &name[7];      // Norman

}
